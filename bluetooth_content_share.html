async function sha256(s){const b=await crypto.subtle.digest('SHA-256',new TextEncoder().encode(s));return Array.from(new Uint8Array(b)).map(x=>x.toString(16).padStart(2,'0')).join('')}
function uid(){return 'id_'+Date.now()+'_'+Math.random().toString(36).slice(2,7)}
function toast(msg,icon='✅'){
const t=document.getElementById('toast');
document.getElementById('toast-msg').textContent=msg;
document.getElementById('toast-icon').textContent=icon;
t.classList.add('show');
clearTimeout(t._t);t._t=setTimeout(()=>t.classList.remove('show'),3000);
}

async function setShared(key,val){try{await window.storage.set(key,JSON.stringify(val),true)}catch(e){console.error(e)}}
async function getPersonal(key){try{const r=await window.storage.get(key,false);return r?JSON.parse(r.value):null}catch{return null}}
async function setPersonal(key,val){try{await window.storage.set(key,JSON.stringify(val),false)}catch(e){console.error(e)}}
